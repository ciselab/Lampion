TABLE info {
    info_key varchar [not null]
    info_value varchar
}  

TABLE positions {
    ROWID int
    simple_class_name varchar [not null]                    
    fully_qualified_class_name varchar [not null]           
    file_name varchar                                    
    method_name varchar                                  
    full_method_name varchar
    Indexes {
        (simple_class_name,fully_qualified_class_name,file_name,method_name,full_method_name) [name:'iPositions']
        id [unique]
    }
}

TABLE transformation_names {
    ROWID int
    transformation_name varchar [unique,not null]
}

TABLE transformation_categories {
    ROWID int
    category_name varchar [unique,not null]
}

TABLE transformations {
    ROWID int
    name_reference int [not null]
    position_reference int [not null]
}

Ref: transformations.position_reference > positions.ROWID

    ROWID int

}
